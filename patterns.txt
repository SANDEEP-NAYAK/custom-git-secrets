
# Passwords and Secret Keys
(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{8,}

# API Keys and Tokens
api[_-]?key[=\s:][\w\d]+
\b(?:[A-Za-z0-9_-]+\.){2}[A-Za-z0-9_-]+\b

# Database Connection Strings
(?i)jdbc:.*?://.*?:.*?@.*?/
(?i)postgresql://.*?:.*?@.*?/

# Private Keys
-----BEGIN [A-Z]+ PRIVATE KEY-----

# Credit Card Numbers
\b(?:\d{4}[-\s]?){3}\d{4}\b

# Email Addresses
\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}\b
